🖥️ Quick Athena Queries

Run in Athena Query Editor → Workgroup: search-keyword-analyzer-dev · Database: search_keyword_analyzer_dev 

-- Top keywords by revenue
SELECT search_engine_domain, search_keyword, revenue
FROM search_keyword_analyzer_dev.adobe_gold
ORDER BY revenue DESC;

-- Hit counts by page (masked bronze — developer accessible)
SELECT pagename, COUNT(*) AS hits
FROM search_keyword_analyzer_dev.adobe_bronze_masked
GROUP BY pagename ORDER BY hits DESC;

-- Purchase events with pseudonymized IP
SELECT date_time, ip, geo_city, product_list
FROM search_keyword_analyzer_dev.adobe_bronze_masked
WHERE event_list LIKE '%1%';

➕ Adding a New Data Source

# Step 1: Create src/salesforce_handler.py (copy adobe_handler.py, update transformation)

# Step 2: Add to terraform/main.tf:
module "salesforce_pipeline" {
  source         = "./modules/pipeline"
  source_name    = "salesforce"
  lambda_handler = "salesforce_handler.lambda_handler"
  bronze_columns = [
    { name = "contact_id", type = "string", comment = "" },
    { name = "revenue",    type = "double",  comment = "" },
  ]
  gold_columns = [
    { name = "campaign", type = "string", comment = "" },
    { name = "revenue",  type = "double",  comment = "" },
  ]
  # All infra vars identical to adobe_pipeline block
  project_name          = var.project_name
  ...
}

# Step 3: Push PR → CI runs terraform apply
# Result: new Lambda + Glue tables + EventBridge rule — no manual steps
